{
  "require":
  {
  	"handlers": ["setUseTimer", "nextFullSecond"],
  	"variables": ["mUseTimer"],
    "mUseTimer":
    {
      "is": "variable",
      "type": "Boolean"
    }
  }
}

public handler OnTimer()
	if mUseTimer then
		redraw all
		schedule timer in nextFullSecond() seconds
	else
		cancel timer
	end if
end handler

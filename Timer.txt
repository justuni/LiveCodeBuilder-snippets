require:
{
  variable: {name: "mUseTimer", type: "Boolean"},
  handler: {name: "nextFullSecond"},
  property: {name: "useTimer", set: "setUseTimer"},
  handler: {name: "setUseTimer"}
}

public handler OnTimer()
	if mUseTimer then
		redraw all
		schedule timer in nextFullSecond() seconds
	else
		cancel timer
	end if
end handler
